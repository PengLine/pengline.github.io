TaskPool 类

即时任务池，专注于立即执行的任务。

初始化

class TaskPool:
    def __init__(
        self,
        executor: Union[Callable, None] = None,
        config: Optional[TaskPoolConfig] = None
    )

参数

参数类型说明
executorCallable任务执行函数（异步函数）
configTaskPoolConfig配置对象

任务提交

submit

def submit(
    self,
    data: Dict[str, Any],
    task_id: Optional[str] = None,
    priority: Union[int, TaskPriority] = TaskPriority.NORMAL,
    delay: float = 0,
    ttl: int = 3600
) -> str

提交即时任务（同步）。

参数类型说明
dataDict任务数据
task_idstr任务ID（可选，自动生成）
priorityint/TaskPriority优先级（0-3，默认2）
delayfloat延迟执行时间（秒）
ttlint任务超时时间（秒）

返回： str - 任务ID

submit_async

async def submit_async(
    self,
    data: Dict[str, Any],
    task_id: Optional[str] = None,
    priority: Union[int, TaskPriority] = TaskPriority.NORMAL,
    delay: float = 0,
    ttl: int = 3600
) -> str

提交即时任务（异步）。

submit_batch / submit_batch_async

def submit_batch(
    self,
    tasks: List[Dict[str, Any]],

) -> List[str]

async def submit_batch_async(
    self,
    tasks: List[Dict[str, Any]],
    priority: Union[int, TaskPriority] = TaskPriority.NORMAL

批量提交任务。

任务等待

wait_for_result

def wait_for_result(
    self,
    task_id: str,
    timeout: float = 300
) -> Any

同步等待任务完成并返回结果。

参数类型说明
task_idstr任务ID
timeoutfloat超时时间（秒）

返回： Any - 任务结果

异常： TaskNotFoundError, TimeoutError, Exception

def wait_all(

等待所有任务完成（同步）。

submit_cron

def submit_cron(
    self,
    data: Dict[str, Any],
    cron_expr: str,
    task_id: Optional[str] = None,
    priority: Union[int, TaskPriority] = TaskPriority.NORMAL
) -> str

按 Cron 表达式周期执行。

Cron 格式说明

字段范围示例
分钟0-590 表示整点
小时0-239 表示上午9点
日1-31* 表示每天
月1-12* 表示每月
周0-6（0=周日）* 表示每周
 
# 每天 9:00 执行
scheduler.submit_cron(data, "0 9 * * *")

# 每 5 分钟执行
scheduler.submit_cron(data, "*/5 * * * *")

# 每周一 9:00 执行
scheduler.submit_cron(data, "0 9 * * 1")

枚举类型

TaskStatus

from neotask import TaskStatus

TaskStatus.PENDING     # "pending"   - 等待执行
TaskStatus.RUNNING     # "running"   - 执行中
TaskStatus.SUCCESS     # "success"   - 执行成功
TaskStatus.FAILED      # "failed"    - 执行失败
TaskStatus.CANCELLED   # "cancelled" - 已取消
TaskStatus.SCHEDULED   # "scheduled" - 已调度（定时任务）

状态流转

# 正常流程
PENDING → RUNNING → SUCCESS

# 失败流程
PENDING → RUNNING → FAILED → PENDING（重试）→ SUCCESS

# 取消流程
PENDING → CANCELLED
RUNNING → CANCELLED

TaskPriority

from neotask import TaskPriority

TaskPriority.CRITICAL  # 0 - 最高优先级
TaskPriority.HIGH      # 1 - 高优先级
TaskPriority.NORMAL    # 2 - 普通优先级（默认）
TaskPriority.LOW       # 3 - 低优先级

配置类

TaskPoolConfig

from neotask import TaskPoolConfig

config = TaskPoolConfig(
    storage_type: str = "memory",      # memory/sqlite/redis
    sqlite_path: str = "neotask.db",   # SQLite 路径
    redis_url: Optional[str] = None,   # Redis URL
    worker_concurrency: int = 10,      # 并发 Worker 数
    prefetch_size: int = 20,           # 预取数量
    queue_max_size: int = 10000,       # 队列最大长度
    max_retries: int = 3,              # 最大重试次数
    retry_delay: float = 1.0,          # 重试延迟（秒）
    task_timeout: Optional[float] = None,  # 任务超时
    enable_metrics: bool = True,       # 启用指标收集
)

SchedulerConfig

from neotask import SchedulerConfig

config = SchedulerConfig(
    storage_type: str = "memory",

    worker_concurrency: int = 10,
    max_retries: int = 3,
    retry_delay: float = 1.0,
    enable_persistence: bool = False,
    scan_interval: float = 0.05,       # 调度扫描间隔（秒）
)

异常类

from neotask.common.exceptions import (
    TaskSchedulerError,      # 基类异常
    TaskNotFoundError,       # 任务不存在
    TaskAlreadyExistsError,  # 任务已存在
    QueueFullError,          # 队列已满
    TimeoutError,            # 等待超时
)

示例

from neotask.common.exceptions import TaskNotFoundError, TimeoutError

try:
    result = pool.wait_for_result(task_id, timeout=30)
except TaskNotFoundError:
    print("任务不存在")
except TimeoutError:
    print("等待超时")
except Exception as e:
    print(f"任务失败: {e}")

事件系统

TaskEvent

class TaskEvent:
    event_type: str    # 事件类型
    task_id: str       # 任务ID
    data: Any          # 事件数据

事件类型

类型说明data 内容
task.created任务创建任务字典
task.started任务开始{"node_id": "..."}
task.completed任务完成{"result": ...}
task.failed任务失败{"error": "..."}
task.cancelled任务取消None

使用示例

async def on_created(event):
    print(f"任务创建: {event.task_id}")

async def on_completed(event):
    print(f"任务完成: {event.task_id}, 结果: {event.data}")

pool.on_created(on_created)
pool.on_completed(on_completed)
